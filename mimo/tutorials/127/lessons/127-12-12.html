If we feel like it, we can also use a computed property for the description.

enumSeason{
  case spring, summer, fall, winter
  var description: String {
    switchself {
    case .summer:
      return"hot season"
    case .winter:
      return"cold season"
    case .spring, .fall:
      return"transitional season"
    }
  }
}
var season = Season.springsummerfallwinter
print(season.description)
